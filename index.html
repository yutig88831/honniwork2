        function updateLunarDisplay() {

            const y = parseInt(document.getElementById('sYear').value);

            const m = parseInt(document.getElementById('sMonth').value);

            const d = parseInt(document.getElementById('sDay').value);

            const label = document.getElementById('lunarResultText');

            if (y > 1900 && m > 0 && m <= 12 && d > 0 && d <= 31) {

                try {

                    const solar = Solar.fromYmd(y, m, d);

                    const lunar = solar.getLunar();

                    label.innerText = `${lunar.getYear() - 1911}.${lunar.getMonth()}.${lunar.getDay()}`;

                } catch(e) { label.innerText = "--"; }

        function handleCheckout() {

            const idVal = document.getElementById('idCard').value.trim();

            const y = document.getElementById('sYear').value;

            const phone = document.getElementById('phone').value.trim();

 

            if (!idVal || !y || !phone) {

            const btn = document.getElementById('submitBtn');

            const now = new Date();

            const tradeDate = now.getFullYear() + '/' +

                             String(now.getMonth() + 1).padStart(2, '0') + '/' +

                             String(now.getDate()).padStart(2, '0') + ' ' +

                             String(now.getHours()).padStart(2, '0') + ':' +

                             String(now.getMinutes()).padStart(2, '0') + ':' +

                             String(now.getSeconds()).padStart(2, '0');

           

            const tradeNo = 'YJ' + Date.now(); // 產生唯一交易編號

 

            const params = {

                MerchantID: ECPAY_CONFIG.MerchantID,

                MerchantTradeNo: tradeNo,

                MerchantTradeDate: tradeDate,

                PaymentType: 'aio',

                TotalAmount: '299', // 設定金額

                TradeDesc: '數字易經專屬解析報告',

                ItemName: '數字易經深度解析報告',

        }

 

        // 綠界檢查碼算法

        function generateCheckMacValue(params, key, iv) {

            let raw = `HashKey=${key}&`;

            const sortedKeys = Object.keys(params).sort();

            raw += sortedKeys.map(k => `${k}=${params[k]}`).join('&');

            raw += `&HashIV=${iv}`;

 

            let encoded = encodeURIComponent(raw).toLowerCase()

                .replace(/%20/g, '+')

                .replace(/%2d/g, '-')

                .replace(/%5f/g, '_')

                .replace(/%2e/g, '.')

                .replace(/%21/g, '!')

                .replace(/%2a/g, '*')

                .replace(/%28/g, '(')

                .replace(/%29/g, ')');

 

            return CryptoJS.SHA256(encoded).toString(CryptoJS.enc.Hex).toUpperCase();

        }
